### 01. 성적 계산
~~~c
#include <stdio.h>
int main(void)
{
 double a=0.3, b=0.3, c=0.4;
 int d=86,e=90,f=86;

 scanf("%lf %d", &a, &d);
~~~

### 02. 30분 전
~~~c
#include <stdio.h>
main(void)
{
 int hour, min;
 printf("시간과 분을 입력하세요 : ");
 scanf("%d %d", &hour, &min);
 printf("입력한 시간의 30분 전 시간은");
 if(min>=30)
  printf("%d시 %d분/n", hour, min-30);
  else
  {
    if(hour==0)
     printf("%d시 %d분/n", 23, min+30);
   else
    printf("%d시 %d분/n", hour-1, min+30);
  }
 return 0;
}
~~~

### 04. 가위바위보
~~~c

int main() {
  char a,b;
  printf("가위 바위 보 중 하나를 선택하시오\n");
  scanf("%s", &a);
  if (a==b) {
   printf("비겼다");
~~~
